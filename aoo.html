let pressTimer; function startPress() { pressTimer = window.setTimeout(function() { // Здесь выполняется код при удержании элемента console.log('Элемент был удержан!'); }, 1000); // Например, удержание должно длиться 1 секунду } function endPress() { clearTimeout(pressTimer); } const element = document.getElementById('yourElementId'); element.addEventListener('touchstart', startPress); element.addEventListener('touchend', endPress); element.addEventListener('touchcancel', endPress); // Для обработки случая, когда касание было прервано
